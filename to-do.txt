move: function (dir, by) {
    if (dir.charAt(0) === 'n') this.y -= by;
    if (dir.charAt(0) === 's') this.y += by;
    if (dir === 'e' || dir.charAt(1) === 'e') this.x += by;
    if (dir === 'w' || dir.charAt(1) === 'w') this.x -= by;

    return this;
},
